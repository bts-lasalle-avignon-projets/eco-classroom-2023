508  public void executerRequete(final String requete)
509  {
510  if(!BaseDeDonnees.active)
511  return;
512 
513  if(estConnecte())
514  {
515  // Seulement pour les requêtes UPDATE, INSERT ou DELETE
516  if(!requete.contains("UPDATE") && !requete.contains("INSERT") &&
517  !requete.contains("DELETE"))
518  return;
519 
520  // Exécute la requête dans un thread
521  Thread requeteBDD = new Thread(new Runnable() {
522  public void run()
523  {
524  mutex.lock();
525  try
526  {
527  Log.d(TAG, "executerRequete() requete = " + requete);
528  Statement statement = connexion.createStatement();
529  int nb = statement.executeUpdate(requete);
530  if(nb > 0)
531  {
532  Message message = new Message();
533  message.what = REQUETE_SQL_OK;
534  message.obj = Integer.toString(nb);
535  if(handler != null)
536  handler.sendMessage(message);
537  }
538  else
539  {
540  Message message = new Message();
541  message.what = REQUETE_SQL_ERREUR;
542  message.obj = "";
543  if(handler != null)
544  handler.sendMessage(message);
545  }
546  }
547  catch(Exception e)
548  {
549  // e.printStackTrace();
550  Log.e(TAG, "executerRequete() Exception = " + e.toString());
551  }
552  finally
553  {
554  mutex.unlock();
555  }
556  }
557  });
558 
559  // Démarrage de l'exécution de la requête
