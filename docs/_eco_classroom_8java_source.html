267  public void filtrerSalles(int choix)
268  {
269  if(salles != null)
270  Log.d(TAG,
271  "filtrerSalles() choix : " + choix + " Nb salles totales : " + salles.size());
272  if(salles == null || salles.size() == 0)
273  return;
274  Vector<Salle> sallesFiltrees = new Vector<Salle>();
275  switch(choix)

277  case TOUTES:
278  sallesFiltrees.addAll(salles);
279  break;
280  case SALLES_LIBRES:
281  for(int indice = 0; indice < salles.size(); indice++)
282  {
283  Salle salle = salles.get(indice);
284  if(!salle.getEstOccupe())
285  sallesFiltrees.add(salle);
286  }
287  break;
288  case DEPASSEMENT_DE_SEUIL:
289  for(int indice = 0; indice < salles.size(); indice++)
290  {
291  Salle salle = salles.get(indice);
292  if(salle.estSeuilTemperatureDepasse() || salle.estSeuilHumiditeDepasse() ||
293  salle.estSeuilCo2Depasse())
294  sallesFiltrees.add(salle);
295  }
296  break;
297  case INTERVENTION:
298  for(int indice = 0; indice < salles.size(); indice++)
299  {
300  Salle salle = salles.get(indice);
301  if((salle.getEtatFenetre() || salle.getEtatLumiere()) && !salle.getEstOccupe())
302  sallesFiltrees.add(salle);
303  }
304  break;
305  default:
306  return;
307  }
308  Log.d(TAG,
309  "filtrerSalles() choix : " + choix +
310  " Nb salles filtrées : " + sallesFiltrees.size());

437  public void traiterMessageMQTT(String topicMQTT, String messageMQTT)
438  {
439  String nomSalle = null;

441  String grandeur = null;
442 
443  Pattern topicPattern = Pattern.compile("(.+?)/(.+?)/(.+?)/(.+)");
444  Matcher topicMatcher = topicPattern.matcher(topicMQTT);
445  if(topicMatcher.find())
446  {
447  nomSalle = topicMatcher.group(2);
448  typeModule = topicMatcher.group(3);
449  grandeur = topicMatcher.group(4);
450  Log.d(TAG,
451  "nomSalle : " + nomSalle + "; typeModule : " + typeModule +
452  "; grandeur : " + grandeur);
453  }
454 
455  Salle.Grandeur g = Salle.retournerGrandeur(grandeur);
456  for(int i = 0; i < salles.size(); ++i)

459  {

463  salles.elementAt(i).setTemperature(Double.valueOf(messageMQTT));
464  this.adaptateurSalle.notifyDataSetChanged();
465  alerterDepassementSeuil(salles.elementAt(i), g);
466  Log.d(TAG,
467  "salle : " + salles.elementAt(i).getNom() +
468  "; temperature : " + salles.elementAt(i).getTemperature());
469  break;
470  case HUMIDITE:
471  salles.elementAt(i).setHumidite(Integer.valueOf(messageMQTT));
472  this.adaptateurSalle.notifyDataSetChanged();
473  alerterDepassementSeuil(salles.elementAt(i), g);
474  Log.d(TAG,
475  "salle : " + salles.elementAt(i).getNom() +
476  "; humidité : " + salles.elementAt(i).getHumidite());
477  break;
478  case CO2:

480  this.adaptateurSalle.notifyDataSetChanged();
481  alerterDepassementSeuil(salles.elementAt(i), g);
482  Log.d(TAG,

484  "; CO2 : " + salles.elementAt(i).getCo2());
485  break;
486  case PRESENCE:
487  if(messageMQTT.compareTo("0") == 0)
488  salles.elementAt(i).setEstOccupe(false);
