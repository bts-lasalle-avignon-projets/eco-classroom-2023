240  public void filtrerSalles(int choix)
241  {
242  if(salles != null)
243  Log.d(TAG,
244  "filtrerSalles() choix : " + choix + " Nb salles totales : " + salles.size());
245  if(salles == null || salles.size() == 0)
246  return;
247  Vector<Salle> sallesFiltrees = new Vector<Salle>();
248  switch(choix)
249  {
250  case TOUTES:
251  sallesFiltrees.addAll(salles);
252  break;
253  case SALLES_LIBRES:
254  for(int indice = 0; indice < salles.size(); indice++)
255  {
256  Salle salle = salles.get(indice);
257  if(!salle.getEstOccupe())
258  sallesFiltrees.add(salle);
259  }
260  break;
261  case DEPASSEMENT_DE_SEUIL:
262  for(int indice = 0; indice < salles.size(); indice++)
263  {
264  Salle salle = salles.get(indice);
265  if(salle.estSeuilTemperatureDepasse() || salle.estSeuilHumiditeDepasse() ||
266  salle.estSeuilCo2Depasse())
267  sallesFiltrees.add(salle);
268  }
269  break;
270  case INTERVENTION:
271  for(int indice = 0; indice < salles.size(); indice++)
272  {
273  Salle salle = salles.get(indice);
274  if((salle.getEtatFenetre() || salle.getEtatLumiere()) && !salle.getEstOccupe())
275  sallesFiltrees.add(salle);
276  }
277  break;
278  default:
279  return;
280  }
281  Log.d(TAG,
282  "filtrerSalles() choix : " + choix +
283  " Nb salles filtrées : " + sallesFiltrees.size());
284  afficherSalles(sallesFiltrees);
285  }
