265 {

267  << salle.getTemperature() << "humidite" << salle.getHumidite()
268  << "CO2" << salle.getCO2() << "lumiere" << salle.getLumiere()
269  << "fenetre" << salle.getFenetre() << "occupation"
270  << salle.getOccupation();
271 
272  // créer des élements de cellule
273  QTableWidgetItem *elementNom, *elementTHI, *elementCO2;
274  QLabel * elementLumiere, *elementFenetre, *elementOccupation;
275 
276  elementNom = new QTableWidgetItem(salle.getNom());
277  elementTHI = new QTableWidgetItem(QString("Inconnu"));
278  elementCO2 = new QTableWidgetItem(QString::number(0));
279  elementLumiere = new QLabel(this);
280  if(salle.getLumiere())
281  elementLumiere->setPixmap(QPixmap(":/images/led-rouge"));
282  else
283  elementLumiere->setPixmap(QPixmap(":/images/led-verte"));
284  elementFenetre = new QLabel(this);
285  if(salle.getFenetre())
286  elementFenetre->setPixmap(QPixmap(":/images/led-rouge"));
287  else
288  elementFenetre->setPixmap(QPixmap(":/images/led-verte"));
289  elementOccupation = new QLabel(this);
290  if(salle.getOccupation())
291  elementOccupation->setPixmap(QPixmap(":/images/led-rouge"));
292  else
293  elementOccupation->setPixmap(QPixmap(":/images/led-verte"));
294 
295  // personnaliser les éléments
296  elementNom->setFlags(Qt::ItemIsEnabled);
297  elementNom->setTextAlignment(Qt::AlignHCenter | Qt::AlignVCenter);
298  elementTHI->setFlags(Qt::ItemIsEnabled);
299  elementTHI->setTextAlignment(Qt::AlignHCenter | Qt::AlignVCenter);
300  elementCO2->setFlags(Qt::ItemIsEnabled);
301  elementCO2->setTextAlignment(Qt::AlignHCenter | Qt::AlignVCenter);
302  elementLumiere->setAlignment(Qt::AlignHCenter | Qt::AlignVCenter);

305 
306  // ajouter une ligne

308  ++nb;
309  tableWidgetSalles->setRowCount(nb);
310 
311  // insérer les éléments de cellule
312  tableWidgetSalles->setItem(nb - 1, COLONNE_SALLE_NOM, elementNom);
313  tableWidgetSalles->setItem(nb - 1,
314  COLONNE_SALLE_CONFORT_THERMIQUE,
315  elementTHI);
316  tableWidgetSalles->setItem(nb - 1, COLONNE_SALLE_QUALITE_AIR, elementCO2);
317  tableWidgetSalles->setCellWidget(nb - 1,
318  COLONNE_SALLE_LUMIERES,
319  elementLumiere);
320  tableWidgetSalles->setCellWidget(nb - 1,
321  COLONNE_SALLE_FENETRES,
322  elementFenetre);
323  tableWidgetSalles->setCellWidget(nb - 1,
324  COLONNE_SALLE_OCCUPATION,
325  elementOccupation);
326 
327  // redimensionner la hauteur de la table
328  tableWidgetSalles->setFixedHeight(
329  tableWidgetSalles->verticalHeader()->length() +
