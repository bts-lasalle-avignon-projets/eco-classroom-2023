310 void Salle::determinerIndiceQualiteAir()
311 {
312  unsigned int valeurCO2 = getCO2();
313  if(valeurCO2 <= SEUIL_HAUT_QUALITE_AIR_EXCELLENTE)
314  indiceQualiteAir = IndiceQualiteAir::Excellente;
315  else if(valeurCO2 >= SEUIL_BAS_QUALITE_AIR_TRES_BIEN &&
316  valeurCO2 <= SEUIL_HAUT_QUALITE_AIR_TRES_BIEN)
317  indiceQualiteAir = IndiceQualiteAir::Tres_Bien;
318  else if(valeurCO2 >= SEUIL_BAS_QUALITE_AIR_MODERE &&
319  valeurCO2 <= SEUIL_HAUT_QUALITE_AIR_MODERE)
320  indiceQualiteAir = IndiceQualiteAir::Modere;
321  else if(valeurCO2 >= SEUIL_BAS_QUALITE_AIR_MAUVAIS &&
322  valeurCO2 <= SEUIL_HAUT_QUALITE_AIR_MAUVAIS)
323  indiceQualiteAir = IndiceQualiteAir::Mauvais;
324  else if(valeurCO2 >= SEUIL_BAS_QUALITE_AIR_TRES_MAUVAIS &&
325  valeurCO2 <= SEUIL_HAUT_QUALITE_AIR_TRES_MAUVAIS)
326  indiceQualiteAir = IndiceQualiteAir::Tres_Mauvais;
327  else
328  indiceQualiteAir = IndiceQualiteAir::Severe;
329 }

331 void Salle::determinerIndiceTHI()
332 {
333  double thom = mesures->calculerTHI();
334  qDebug() << Q_FUNC_INFO << "thom" << thom;
335  if(thom < SEUIL_THOM_FROID)
336  {
337  indiceConfortTHI = IndiceTHI::Froid;
338  }
339  else if(thom >= SEUIL_THOM_FROID && thom < SEUIL_THOM_FRAIS)
340  {
341  indiceConfortTHI = IndiceTHI::Frais;
342  }
343  else if(thom >= SEUIL_THOM_FRAIS && thom < SEUIL_THOM_LEGEREMENT_FRAIS)
344  {
345  indiceConfortTHI = IndiceTHI::LegerementFrais;
346  }
347  else if(thom >= SEUIL_THOM_LEGEREMENT_FRAIS && thom < SEUIL_THOM_NEUTRE)
348  {
349  indiceConfortTHI = IndiceTHI::Neutre;
350  }
351  else if(thom >= SEUIL_THOM_NEUTRE && thom < SEUIL_THOM_LEGEREMENT_TIEDE)
352  {
353  indiceConfortTHI = IndiceTHI::Tiede;
354  }
355  else if(thom >= SEUIL_THOM_LEGEREMENT_TIEDE && thom < SEUIL_THOM_TIEDE)
356  {
357  indiceConfortTHI = IndiceTHI::LegerementTiede;
358  }
359  else if(thom >= SEUIL_THOM_TIEDE)
360  {
361  indiceConfortTHI = IndiceTHI::Chaud;
362  }
363  else
364  indiceConfortTHI = IndiceTHI::Inconnu;
365  qDebug() << Q_FUNC_INFO << "indiceConfortTHI" << indiceConfortTHI
366  << getTHI();
